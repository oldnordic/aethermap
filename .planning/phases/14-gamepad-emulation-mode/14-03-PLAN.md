```rust
pub async fn process_as_gamepad(
    &self,
    device_id: &str,
    raw_x: i32,
    raw_y: i32,
) -> Option<(i32, i32)> {
    use crate::analog_calibration::{AnalogCalibration, DeadzoneShape, SensitivityCurve};

    // Get device config
    let config = self.devices.read().await.get(device_id).copied()?;

    // Build calibration from device config
    let calibration = AnalogCalibration {
        deadzone_shape: DeadzoneShape::Circular,
        deadzone_size: config.inner_deadzone_x as f32 / MAX_ABS_VALUE as f32,
        sensitivity_curve: SensitivityCurve::Linear,
        sensitivity_multiplier: config.sensitivity,
        range_min: -32768.0,
        range_max: 32767.0,
        invert_x: false,
        invert_y: false,
    };

    // Normalize raw ADC (0-255) to 0.0-1.0
    let nx = raw_x as f32 / 255.0;
    let ny = raw_y as f32 / 255.0;

    // Center to -0.5 to 0.5
    let cx = nx - 0.5;
    let cy = 0.5 - ny;  // Invert Y (up is negative in screen coords)

    // Calculate magnitude and angle for polar processing
    let magnitude = (cx * cx + cy * cy).sqrt();
    let angle = cy.atan2(cx);

    // Apply deadzone
    let deadzone = calibration.deadzone_size;
    if magnitude < deadzone {
        return None;  // Within deadzone - no output
    }

    // Scale magnitude from deadzone boundary to full range
    let max_magnitude = 0.707;  // sqrt(0.5) for circular deadzone
    let scaled_magnitude = (magnitude - deadzone) / (max_magnitude - deadzone);

    // Apply sensitivity
    let adjusted_magnitude = match calibration.sensitivity_curve {
        SensitivityCurve::Linear => scaled_magnitude * calibration.sensitivity_multiplier,
        SensitivityCurve::Quadratic => scaled_magnitude.powi(2) * calibration.sensitivity_multiplier,
        SensitivityCurve::Exponential { exponent } => scaled_magnitude.powf(exponent) * calibration.sensitivity_multiplier,
    };

    // Clamp and convert back to Cartesian
    let clamped = adjusted_magnitude.min(1.0).max(0.0);
    let out_x = angle.cos() * clamped;
    let out_y = angle.sin() * clamped;

    // Scale to Linux input range
    let range = calibration.range_max - calibration.range_min;
    let x = ((out_x * range) + calibration.range_min) as i32;
    let y = ((out_y * range) + calibration.range_min) as i32;

    Some((x, y))
}
```

```rust
async fn test_process_as_gamepad_center() {
    let processor = AnalogProcessor::new();

    // Center position (127, 127) should be filtered by deadzone
    let result = processor.process_as_gamepad(device_id, 127, 127).await;
    assert!(result.is_none(), "Center position should be filtered by deadzone");
}

#[tokio::test]
async fn test_process_as_gamepad_full_deflection() {
    let processor = AnalogProcessor::new();
    let device_id = "32b6:12f7";

    // Full right (255, 127)
    if let Some((x, y)) = processor.process_as_gamepad(device_id, 255, 127).await {
        assert!(x > 30000, "X should be near max positive");
        assert!(y.abs() < 5000, "Y should be near center");
    } else {
        panic!("Full right deflection should produce output");
    }

    // Full up (127, 0)
    if let Some((x, y)) = processor.process_as_gamepad(device_id, 127, 0).await {
        assert!(x.abs() < 5000, "X should be near center");
        assert!(y < -30000, "Y should be near max negative (up)");
    } else {
        panic!("Full up deflection should produce output");
    }
}
```
