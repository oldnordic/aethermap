async fn test_process_as_gamepad_sensitivity_affects_output() {
    let processor = AnalogProcessor::new();
    let device_id = "test_sens";

    // Set high sensitivity
    {
        let mut devices = processor.devices.write().await;
        devices.insert(device_id.to_string(), DeviceAnalogConfig {
            device_id: device_id.to_string(),
            sensitivity: 2.0,  // 2x sensitivity
            ..Default::default()
        });
    }

    let result = processor.process_as_gamepad(device_id, 200, 127).await;
    assert!(result.is_some());

    // With 2x sensitivity, output should be higher than default
    let (x, _) = result.unwrap();
    assert!(x > 0, "X should be positive");
}

mod gamepad_tests {

        let mut layer_state = layer_manager.create_device_state("32b6:12f7".to_string());
        layer_state.layer_configs[1].analog_mode = AnalogMode::Gamepad;
        layer_manager.add_device_state(layer_state).await;

        // Activate layer 1
        layer_manager.activate_hold_layer("32b6:12f7", 1).await.unwrap();

        // Create mock device manager (simplified for testing)

        // This is a structural test showing the integration pattern

        assert_eq!(effective_layer, 1, "Layer 1 should be effective");

        let layer_config = layer_manager.get_layer_config("32b6:12f7", 1).await;
        assert!(layer_config.is_some());
        assert_eq!(layer_config.unwrap().analog_mode, AnalogMode::Gamepad);
    }

    #[tokio::test]
    async fn test_per_layer_analog_calibration() {
        use crate::analog_calibration::{AnalogCalibration, DeadzoneShape, SensitivityCurve};

        // Layer 0: default calibration
        // Layer 1: custom calibration with quadratic curve

        // Verify layer 1 has custom calibration

        assert!(layer_config.is_some());

        let cal = layer_config.unwrap().analog_calibration;
        assert!(cal.is_some());
