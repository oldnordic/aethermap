```rust
/// Process analog stick event for gamepad output
///
/// This method is called from the device event loop when an analog stick
/// event is detected and the active layer's analog_mode is Gamepad.
///
/// # Arguments
///
/// * `device_id` - Device identifier (vendor:product)
/// * `raw_x` - Raw X axis value (0-255)
/// * `raw_y` - Raw Y axis value (0-255)
///
/// # Returns
///
/// * `Ok(())` - Event processed successfully
/// * `Err(String)` - Processing failed
pub async fn process_analog_gamepad(
    &self,
    device_id: &str,
    raw_x: i32,
    raw_y: i32,
) -> Result<(), String> {
    // Get the active layer for this device
    let effective_layer = self.layer_manager
        .get_effective_layer(device_id)
        .await
        .unwrap_or(0);

    // Get the layer config to verify analog_mode is Gamepad
    let layer_config = self.layer_manager
        .get_layer_config(device_id, effective_layer)
        .await
        .ok_or("Layer config not found")?;

    if layer_config.analog_mode != AnalogMode::Gamepad {
        return Err("Layer analog_mode is not Gamepad".to_string());
    }

    // Use layer-specific calibration if configured, otherwise use device default
    let calibration = layer_config.analog_calibration
        .as_ref()
        .map(|cal| {
            // Use layer calibration
            // Convert AnalogCalibration to device config format
            // This is a simplified version - full integration in task 3
            None
        });

    // Process through calibration pipeline
    let processor = self.analog_processor.as_ref()
        .ok_or("AnalogProcessor not initialized")?;

    if let Some((x, y)) = processor.process_as_gamepad(device_id, raw_x, raw_y).await {
        // Emit to virtual gamepad device
        use crate::gamepad_device::GamepadAxis;

        self.gamepad_device.emit_axis(GamepadAxis::X, x).await
            .map_err(|e| format!("Failed to emit X axis: {}", e))?;
        self.gamepad_device.emit_axis(GamepadAxis::Y, y).await
            .map_err(|e| format!("Failed to emit Y axis: {}", e))?;
    }

    Ok(())
}
```
