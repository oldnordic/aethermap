    ```rust
    Request::SetAutoSwitchRules { rules } => {
        debug!("SetAutoSwitchRules: {} rules", rules.len());

        match config_manager.set_auto_switch_rules(rules.clone()).await {
            Ok(()) => {
                // Reload auto-profile switcher
                if let Some(switcher) = auto_profile_switcher.as_ref() {
                    switcher.write().await.load_rules(rules);
                    info!("Reloaded auto-switch rules, {} rules active", rules.len());
                }

                info!("Saved {} auto-switch rules", rules.len());
                Response::AutoSwitchRulesAck
            }
            Err(e) => {
                warn!("SetAutoSwitchRules failed: {}", e);
                Response::Error(format!("Failed to save auto-switch rules: {}", e))
            }
        }
    }
    ```

    Follow SetAnalogCalibration pattern: save to config, reload component, return ack.

    DO NOT forget to call load_rules() on AutoProfileSwitcher to apply new rules.
