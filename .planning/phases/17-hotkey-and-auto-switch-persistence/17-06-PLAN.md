    ```rust
    Message::DeleteHotkeyBinding(index) => {
        if let Some(view) = &self.hotkey_view {
            if index < view.bindings.len() {
                let device_id = view.device_id.clone();
                let binding = view.bindings[index].clone();
                let socket_path = self.socket_path.clone();

                // Update local state immediately
                let bindings = view.bindings.clone();
                let mut updated_bindings = bindings.clone();
                updated_bindings.remove(index);

                return Command::perform(
                    async move {
                        let client = IpcClient::new(socket_path);
                        let request = Request::RemoveHotkey {
                            device_id,
                            key: binding.key,
                            modifiers: binding.modifiers,
                        };
                        match client.send(&request).await {
                            Ok(Response::HotkeyRemoved { .. }) => Ok(()),
                            Ok(Response::Error(msg)) => Err(msg),
                            Err(e) => Err(format!("IPC error: {}", e)),
                            _ => Err("Unexpected response".to_string()),
                        }
                    },
                    move |result| {
                        // Update local state on success
                        if result.is_ok() {
                            // Return message to update view
                            Message::HotkeyBindingsUpdated(updated_bindings)
                        } else {
                            let err_msg = result.unwrap_err();
                            Message::AddNotification(format!("Failed to delete hotkey: {}", err_msg), true)
                        }
                    }
                );
            }
        }
        Command::none()
    }
    ```

    Add new Message variant `HotkeyBindingsUpdated(Vec<HotkeyBinding>)` to handle successful delete.

    DO NOT forget to handle the response to update local view state.
  </action>
