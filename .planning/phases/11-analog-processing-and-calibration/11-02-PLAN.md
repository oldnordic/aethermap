///
/// # Arguments
///
/// * `x` - Raw X value (typically 0-255 for 8-bit ADC)
/// * `y` - Raw Y value (typically 0-255 for 8-bit ADC)
///
/// # Returns
///
/// Calibrated (x, y) in range [range_min, range_max]
pub fn process(&self, x: i32, y: i32) -> (i32, i32) {
    // Step 1: Normalize to 0.0-1.0 range
    let (nx, ny) = self.normalize(x, y);

    // Step 2: Center at origin (0,0) -> (-0.5 to 0.5)
    let (cx, cy) = self.center(nx, ny);

    // Step 3: Apply deadzone filter
    let (dx, dy) = self.apply_deadzone(cx, cy);

    // Step 4: Apply sensitivity curve
    let (sx, sy) = self.apply_sensitivity(dx, dy);

    // Step 5: Scale to output range with inversion
    let (ox, oy) = self.scale_to_output(sx, sy);

    (ox, oy)
}

3. **apply_deadzone()** - Filter small movements:
```rust
fn apply_deadzone(&self, x: f32, y: f32) -> (f32, f32) {
    match self.calibration.deadzone_shape {
        DeadzoneShape::Circular => {
            let magnitude = (x * x + y * y).sqrt();
            if magnitude < self.calibration.deadzone {
                return (0.0, 0.0);
            }
            // Scale output to avoid "jump" at deadzone boundary
            let scale = (magnitude - self.calibration.deadzone) / (1.0 - self.calibration.deadzone);
            let angle = y.atan2(x);
            (angle.cos() * magnitude.min(1.0) * scale, angle.sin() * magnitude.min(1.0) * scale)
        }
        DeadzoneShape::Square => {
            let dx = if x.abs() < self.calibration.deadzone { 0.0 } else { x };
            let dy = if y.abs() < self.calibration.deadzone { 0.0 } else { y };
            (dx, dy)
        }
    }
}
```

4. **apply_sensitivity()** - Apply response curve:
```rust
fn apply_sensitivity(&self, x: f32, y: f32) -> (f32, f32) {
    let magnitude = (x * x + y * y).sqrt();
    if magnitude == 0.0 {
        return (0.0, 0.0);
    }
    let angle = y.atan2(x);

    let scaled = match self.calibration.sensitivity {
        SensitivityCurve::Linear => magnitude,
        SensitivityCurve::Quadratic => magnitude * magnitude,
        SensitivityCurve::Exponential => magnitude.powf(3.0),
    } * self.calibration.sensitivity_multiplier;

    let result = scaled.min(1.0);
    (angle.cos() * result, angle.sin() * result)
}
```

5. **scale_to_output()** - Convert to Linux input range:
```rust
fn scale_to_output(&self, x: f32, y: f32) -> (i32, i32) {
    let range = (self.calibration.range_max - self.calibration.range_min) as f32;
    let mut ox = (x * range) as i32 + self.calibration.range_min;
    let mut oy = (y * range) as i32 + self.calibration.range_min;

    if self.calibration.invert_x {
        ox = self.calibration.range_max + self.calibration.range_min - ox;
    }
    if self.calibration.invert_y {
        oy = self.calibration.range_max + self.calibration.range_min - oy;
    }

    // Clamp to valid range
    ox = ox.clamp(self.calibration.range_min, self.calibration.range_max);
    oy = oy.clamp(self.calibration.range_min, self.calibration.range_max);

    (ox, oy)
}
```
