```rust
/// Detect 8-way D-pad direction from normalized (-1.0 to 1.0) coordinates
///
/// Uses 22.5-degree sectors for 8-way direction detection.
/// Applies deadzone before direction checking.
fn detect_dpad_direction(&self, x: f32, y: f32) -> DpadDirection {
    // Apply deadzone first to prevent drift
    let (dx, dy) = self.apply_deadzone(x, y);

    let magnitude = (dx * dx + dy * dy).sqrt();
    if magnitude < 0.1 {
        return DpadDirection::None;  // Too small after deadzone
    }

    // Calculate angle in degrees (0-360, 0 = right, clockwise)
    let angle_rad = dy.atan2(x);
    let angle_deg = angle_rad.to_degrees();
    let normalized_angle = if angle_deg < 0.0 { angle_deg + 360.0 } else { angle_deg };

    // 8 sectors of 45 degrees each, centered on cardinal directions
    // Right: 337.5-22.5, Up: 67.5-112.5, Left: 157.5-202.5, Down: 247.5-292.5
    match normalized_angle {
        a if a >= 337.5 || a < 22.5 => DpadDirection::Right,
        a if a >= 22.5 && a < 67.5 => DpadDirection::DownRight,
        a if a >= 67.5 && a < 112.5 => DpadDirection::Down,
        a if a >= 112.5 && a < 157.5 => DpadDirection::DownLeft,
        a if a >= 157.5 && a < 202.5 => DpadDirection::Left,
        a if a >= 202.5 && a < 247.5 => DpadDirection::UpLeft,
        a if a >= 247.5 && a < 292.5 => DpadDirection::Up,
        a if a >= 292.5 && a < 337.5 => DpadDirection::UpRight,
        _ => DpadDirection::None,
    }
}
```

Note: Angle orientation matches Phase 09-04 hat switch pattern (Y is inverted for standard coordinate system).

/// Convert direction to Linux key codes

fn direction_to_keys(direction: DpadDirection) -> Vec<Key> {
    match direction {
        DpadDirection::None => vec![],
        DpadDirection::Up => vec![Key::KEY_UP],
        DpadDirection::Down => vec![Key::KEY_DOWN],
        DpadDirection::Left => vec![Key::KEY_LEFT],
        DpadDirection::Right => vec![Key::KEY_RIGHT],
        DpadDirection::UpLeft => vec![Key::KEY_UP, Key::KEY_LEFT],
        DpadDirection::UpRight => vec![Key::KEY_UP, Key::KEY_RIGHT],
        DpadDirection::DownLeft => vec![Key::KEY_DOWN, Key::KEY_LEFT],
        DpadDirection::DownRight => vec![Key::KEY_DOWN, Key::KEY_RIGHT],
    }
}

/// Process analog input as D-pad output
///
/// Converts analog stick position to 8-way directional keys.
/// Returns a vector of (key, pressed) tuples for the current state.
///
/// # Arguments
///
/// * `x` - Raw X value (0-255)
/// * `y` - Raw Y value (0-255)
///
/// # Returns
///
/// Vec of (Key, bool) where bool indicates pressed (true) or released (false)
/// Cardinal directions return 1 key, diagonals return 2 keys
///
/// # Examples
///
/// ```ignore
/// let output = processor.process_as_dpad(200, 128);  // Right
/// // Returns: [(KEY_RIGHT, true)]
///
/// let output = processor.process_as_dpad(200, 50);  // Up-Right
/// // Returns: [(KEY_UP, true), (KEY_RIGHT, true)]
/// ```
pub fn process_as_dpad(&self, x: i32, y: i32) -> Vec<(Key, bool)> {
    // Normalize and center the input
    let (nx, ny) = self.normalize(x, y);
    let (cx, cy) = self.center(nx, ny);

    // Scale to -1.0 to 1.0 range for direction detection
    let (sx, sy) = (cx * 2.0, cy * 2.0);

    // Detect direction
    let direction = self.detect_dpad_direction(sx, sy);

    // Convert to key codes (all pressed = true)
    direction_to_keys(direction)
        .into_iter()
        .map(|k| (k, true))
        .collect()
}
```

Note: This method returns only the current frame's state. The caller (device.rs event loop) is responsible for tracking previous state and sending release events when direction changes (matching Phase 09-04 hat switch pattern).
