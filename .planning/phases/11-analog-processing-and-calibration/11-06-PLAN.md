async fn test_analog_macro_recording() {
    let actions = vec![
        MacroAction::analog(100, 128),
        MacroAction::analog(200, 128),
        MacroAction::analog(255, 0),
    ];

    let macro_entry = MacroEntry {
        name: "analog_test".to_string(),
        trigger_keys: vec![Key::KEY_A],
        actions,
        device_id: None,
    };

    // Verify serialize/deserialize works
    let yaml = serde_yaml::to_string(&macro_entry).unwrap();
    let restored: MacroEntry = serde_yaml::from_str(&yaml).unwrap();

    assert_eq!(restored.actions.len(), 3);
    match &restored.actions[0] {
        MacroAction::Analog { x, y } => {
            assert_eq!(*x, 100);
            assert_eq!(*y, 128);
        }
        _ => panic!("Expected Analog action"),
    }
}
