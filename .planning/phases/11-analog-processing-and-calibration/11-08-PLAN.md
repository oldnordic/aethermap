```rust
impl AnalogCalibrationView {
    fn view(&self) -> Element<Message> {
        let content = Column::new()
            .spacing(20)
            .push(text("Analog Calibration").size(24))

            // Device and layer info
            .push(text(format!("Device: {}", self.device_id)).size(14))
            .push(text(format!("Layer: {}", self.layer_id)).size(14))

            // Deadzone section
            .push(rule(1))
            .push(text("Deadzone").size(18))
            .push(
                Row::new()
                    .spacing(10)
                    .push(text("Size:"))
                    .push(text(format!("{:.0}%", self.calibration.deadzone * 100.0)))
                    .push(slider(0.0..=1.0, self.calibration.deadzone, Message::AnalogDeadzoneChanged))
            )
            .push(
                Row::new()
                    .spacing(10)
                    .push(pick_list(
                        &DeadzoneShape::ALL[..],
                        Some(self.deadzone_shape_selected),
                        Message::AnalogDeadzoneShapeChanged,
                    ))
            )

            // Sensitivity section
            .push(rule(1))
            .push(text("Sensitivity").size(18))
            .push(
                Row::new()
                    .spacing(10)
                    .push(text("Multiplier:"))
                    .push(text(format!("{:.1}", self.calibration.sensitivity_multiplier)))
                    .push(slider(0.1..=5.0, self.calibration.sensitivity_multiplier, Message::AnalogSensitivityChanged).step(0.1))
            )
            .push(
                Row::new()
                    .spacing(10)
                    .push(pick_list(
                        &SensitivityCurve::ALL[..],
                        Some(self.sensitivity_curve_selected),
                        Message::AnalogSensitivityCurveChanged,
                    ))
            )

            // Range section
            .push(rule(1))
            .push(text("Output Range").size(18))
            .push(
                Row::new()
                    .spacing(10)
                    .push(text("Min:"))
                    .push(text(self.calibration.range_min.to_string()))
                    .push(slider(-32768..=0, self.calibration.range_min as f32, |v| Message::AnalogRangeMinChanged(v as i32)))
            )
            .push(
                Row::new()
                    .spacing(10)
                    .push(text("Max:"))
                    .push(text(self.calibration.range_max.to_string()))
                    .push(slider(0..=32767, self.calibration.range_max as f32, |v| Message::AnalogRangeMaxChanged(v as i32)))
            )

            // Inversion section
            .push(rule(1))
            .push(text("Axis Inversion").size(18))
            .push(
                Row::new()
                    .spacing(20)
                    .push(checkbox("Invert X", self.invert_x_checked, Message::AnalogInvertXToggled))
                    .push(checkbox("Invert Y", self.invert_y_checked, Message::AnalogInvertYToggled))
            )

            // Apply button
            .push(rule(1))
            .push(
                button("Apply Calibration")
                    .on_press(Message::ApplyAnalogCalibration)
            );

        // Error display
        let content = if let Some(error) = &self.error {
            content.push(text(format!("Error: {}", error)).style(Color::RGB(255, 100, 100)))
        } else {
            content
        };

        scrollable(content).height(Length::Fill).into()
    }
}
```

1. **Add OpenAnalogCalibration handler**:

```rust
Message::OpenAnalogCalibration { device_id, layer_id } => {
    return Command::perform(
        async move {
            ipc_client::get_analog_calibration(&device_id, layer_id).await
                .map_err(|e| e.to_string())
        },
        Message::AnalogCalibrationLoaded,
    );
}
```

2. **Add AnalogCalibrationLoaded handler**:
```rust
Message::AnalogCalibrationLoaded(Ok(calibration)) => {
    // Create or update the calibration view
    self.analog_calibration_view = Some(AnalogCalibrationView {
        device_id: previous_device_id,
        layer_id: previous_layer_id,
        calibration: calibration.clone(),
        // Initialize UI state from calibration
        deadzone_shape_selected: match calibration.deadzone_shape.as_str() {
            "circular" => DeadzoneShape::Circular,
            "square" => DeadzoneShape::Square,
            _ => DeadzoneShape::Circular,
        },
        sensitivity_curve_selected: match calibration.sensitivity.as_str() {
            "linear" => SensitivityCurve::Linear,
            "quadratic" => SensitivityCurve::Quadratic,
            "exponential" => SensitivityCurve::Exponential,
            _ => SensitivityCurve::Linear,
        },
        invert_x_checked: calibration.invert_x,
        invert_y_checked: calibration.invert_y,
        loading: false,
        error: None,
    });
}

Message::AnalogCalibrationLoaded(Err(error)) => {
    if let Some(view) = &mut self.analog_calibration_view {
        view.error = Some(error);
        view.loading = false;
    }
}
```

3. **Add calibration field change handlers** (update local state):

```rust
Message::AnalogDeadzoneChanged(value) => {
    if let Some(view) = &mut self.analog_calibration_view {
        view.calibration.deadzone = value;
    }
}

Message::AnalogDeadzoneShapeChanged(shape) => {
    if let Some(view) = &mut self.analog_calibration_view {
        view.deadzone_shape_selected = shape;
        view.calibration.deadzone_shape = shape.to_string();
    }
}

// ... similar handlers for other fields
```

4. **Add ApplyAnalogCalibration handler**:
```rust
Message::ApplyAnalogCalibration => {
    if let Some(view) = self.analog_calibration_view.as_ref() {
        let device_id = view.device_id.clone();
        let layer_id = view.layer_id;
        let calibration = view.calibration.clone();

        return Command::perform(
            async move {
                ipc_client::set_analog_calibration(&device_id, layer_id, calibration).await
                    .map_err(|e| e.to_string())
            },
            Message::AnalogCalibrationApplied,
        );
    }
}

Message::AnalogCalibrationApplied(Ok(())) => {
    // Success - maybe show confirmation or close view
}

Message::AnalogCalibrationApplied(Err(error)) => {
    if let Some(view) = &mut self.analog_calibration_view {
        view.error = Some(error);
    }
}
```
