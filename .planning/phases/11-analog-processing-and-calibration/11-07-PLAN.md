/// Convert IPC config to internal AnalogCalibration
fn config_to_calibration(config: AnalogCalibrationConfig) -> Result<AnalogCalibration, String> {
    // Validate deadzone range
    if config.deadzone < 0.0 || config.deadzone > 1.0 {
        return Err("Deadzone must be between 0.0 and 1.0".to_string());
    }

    let deadzone_shape = match config.deadzone_shape.as_str() {
        "circular" => DeadzoneShape::Circular,
        "square" => DeadzoneShape::Square,
        _ => return Err(format!("Invalid deadzone shape: {}", config.deadzone_shape)),
    };

    let sensitivity = match config.sensitivity.as_str() {
        "linear" => SensitivityCurve::Linear,
        "quadratic" => SensitivityCurve::Quadratic,
        "exponential" => SensitivityCurve::Exponential,
        _ => return Err(format!("Invalid sensitivity curve: {}", config.sensitivity)),
    };

    Ok(AnalogCalibration {
        deadzone: config.deadzone,
        deadzone_shape,
        sensitivity,
        sensitivity_multiplier: config.sensitivity_multiplier,
        range_min: config.range_min,
        range_max: config.range_max,
        invert_x: config.invert_x,
        invert_y: config.invert_y,
    })
}

/// Convert internal AnalogCalibration to IPC config
fn calibration_to_config(calibration: &AnalogCalibration) -> AnalogCalibrationConfig {
    AnalogCalibrationConfig {
        deadzone: calibration.deadzone,
        deadzone_shape: match calibration.deadzone_shape {
            DeadzoneShape::Circular => "circular".to_string(),
            DeadzoneShape::Square => "square".to_string(),
        },
        sensitivity: match calibration.sensitivity {
            SensitivityCurve::Linear => "linear".to_string(),
            SensitivityCurve::Quadratic => "quadratic".to_string(),
            SensitivityCurve::Exponential => "exponential".to_string(),
        },
        sensitivity_multiplier: calibration.sensitivity_multiplier,
        range_min: calibration.range_min,
        range_max: calibration.range_max,
        invert_x: calibration.invert_x,
        invert_y: calibration.invert_y,
    }
}
```

3. **Add handler in handle_request()** match arm:

Request::GetAnalogCalibration { device_id, layer_id } => {
    let calibration = state
        .config_manager
        .get_analog_calibration(&device_id, layer_id);

    let config = calibration.as_ref().map(calibration_to_config);

    Response::AnalogCalibration {
        device_id,
        layer_id,
        calibration: config,
    }
}

Request::SetAnalogCalibration { device_id, layer_id, calibration } => {
    match config_to_calibration(calibration) {
        Ok(cal) => {
            // Update the grabbed device's calibration
            let mut devices = state.devices.write().await;
            if let Some(device) = devices.get_mut(&device_id) {
                device.update_calibration(Some(cal));
                Response::AnalogCalibrationAck
            } else {
                Response::Error(format!("Device not found: {}", device_id))
            }
        }
        Err(e) => Response::Error(format!("Invalid calibration: {}", e)),
    }
}
```

/// Get analog calibration for a device and layer
///
/// # Arguments
///
/// * `device_id` - Device identifier
/// * `layer_id` - Layer ID (0=base, 1, 2, ...)
///
/// # Returns
///
/// * `Ok(AnalogCalibrationConfig)` - Calibration settings
/// * `Err(IpcError)` - Communication error or device not found
pub async fn get_analog_calibration(
    device_id: &str,
    layer_id: usize,
) -> Result<AnalogCalibrationConfig, IpcError> {
    let request = Request::GetAnalogCalibration {
        device_id: device_id.to_string(),
        layer_id,
    };

    match send(&request).await? {
        Response::AnalogCalibration { calibration: Some(cal), .. } => Ok(cal),
        Response::AnalogCalibration { calibration: None, .. } => {
            // Return default config
            Ok(AnalogCalibrationConfig {
                deadzone: 0.15,
                deadzone_shape: "circular".to_string(),
                sensitivity: "linear".to_string(),
                sensitivity_multiplier: 1.0,
                range_min: -32768,
                range_max: 32767,
                invert_x: false,
                invert_y: false,
            })
        }
        Response::Error(msg) => Err(IpcError::RequestFailed(msg)),
        _ => Err(IpcError::UnexpectedResponse),
    }
}
```

2. **Add set_analog_calibration() method**:

/// Set analog calibration for a device and layer
///
/// # Arguments
///
/// * `device_id` - Device identifier
/// * `layer_id` - Layer ID
/// * `calibration` - New calibration settings
///
/// # Returns
///
/// * `Ok(())` - Calibration updated successfully
/// * `Err(IpcError)` - Communication or validation error
pub async fn set_analog_calibration(
    device_id: &str,
    layer_id: usize,
    calibration: AnalogCalibrationConfig,
) -> Result<(), IpcError> {
    let request = Request::SetAnalogCalibration {
        device_id: device_id.to_string(),
        layer_id,
        calibration,
    };

    match send(&request).await? {
        Response::AnalogCalibrationAck => Ok(()),
        Response::Error(msg) => Err(IpcError::RequestFailed(msg)),
        _ => Err(IpcError::UnexpectedResponse),
    }
}
```

async fn test_analog_calibration_conversion() {
