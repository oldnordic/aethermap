```rust
/// Get analog calibration for a specific device and layer
///
/// # Arguments
///
/// * `device_id` - Device identifier (e.g., "32b6:12f7")
/// * `layer_id` - Layer ID (0=base, 1, 2, ...)
///
/// # Returns
///
/// * `Some(AnalogCalibration)` if configured for this device/layer
/// * `None` if not configured (use AnalogCalibration::default())
pub fn get_analog_calibration(
    &self,
    device_id: &str,
    layer_id: usize,
) -> Option<AnalogCalibration> {
    self.device_profiles.get(device_id)
        .and_then(|profiles| profiles.iter().find(|p| p.analog_calibration.contains_key(&layer_id)))
        .and_then(|profile| profile.analog_calibration.get(&layer_id))
        .cloned()
}

/// Get all analog calibrations for a device (all layers)
///
/// # Returns
///
/// HashMap of layer_id -> AnalogCalibration (empty if none configured)
pub fn get_all_analog_calibrations(
    &self,
    device_id: &str,
) -> HashMap<usize, AnalogCalibration> {
    let mut calibrations = HashMap::new();

    if let Some(profiles) = self.device_profiles.get(device_id) {
        for profile in profiles {
            for (layer_id, calibration) in &profile.analog_calibration {
                calibrations.insert(*layer_id, calibration.clone());
            }
        }
    }

    calibrations
}
```

#[tokio::test]
async fn test_analog_calibration_loading() {
    let temp_dir = TempDir::new().unwrap();
    let config_path = temp_dir.path().join("test.yaml");

    // Create test config with analog calibration
    let yaml = r#"
devices:
  "32b6:12f7":
    analog_calibration:
      0:
        deadzone: 0.15
        sensitivity: linear
      1:
        deadzone: 0.10
        sensitivity: quadratic
"#;

    fs::write(&config_path, yaml).await.unwrap();
    let manager = ConfigManager::load(&config_path).await.unwrap();

    // Verify base layer calibration
    let cal0 = manager.get_analog_calibration("32b6:12f7", 0);
    assert!(cal0.is_some());
    assert_eq!(cal0.unwrap().deadzone, 0.15);

    // Verify layer 1 calibration
    let cal1 = manager.get_analog_calibration("32b6:12f7", 1);
    assert!(cal1.is_some());
    assert_eq!(cal1.unwrap().deadzone, 0.10);

    // Verify non-existent layer returns None
    let cal2 = manager.get_analog_calibration("32b6:12f7", 2);
    assert!(cal2.is_none());
}
