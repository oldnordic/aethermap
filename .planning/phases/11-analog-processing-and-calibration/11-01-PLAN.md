3. **AnalogCalibration struct** (per CONTEXT.md data structures):
```rust
#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(default)]
pub struct AnalogCalibration {
    /// Deadzone radius (0.0 to 1.0)
    /// Values below this are filtered out
    #[serde(default = "default_deadzone")]
    pub deadzone: f32,

    /// Shape of the deadzone
    #[serde(default)]
    pub deadzone_shape: DeadzoneShape,

    /// Sensitivity response curve
    #[serde(default)]
    pub sensitivity: SensitivityCurve,

    /// Sensitivity multiplier (0.1 to 5.0)
    #[serde(default = "default_sensitivity_multiplier")]
    pub sensitivity_multiplier: f32,

    /// Minimum output value (default -32768)
    #[serde(default = "default_range_min")]
    pub range_min: i32,

    /// Maximum output value (default 32767)
    #[serde(default = "default_range_max")]
    pub range_max: i32,

    /// Invert X axis
    #[serde(default)]
    pub invert_x: bool,

    /// Invert Y axis
    #[serde(default)]
    pub invert_y: bool,
}
```
