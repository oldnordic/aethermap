  Request::GetDeviceCapabilities { device_path } => {
      // Query device from DeviceManager
      let device_info = state.device_manager.read().await

      match device_info {
          Some(info) => {
              // Detect capabilities from device
              let capabilities = detect_device_capabilities(&info, &state).await;

              Response::DeviceCapabilities {

  3. Add helper function to detect capabilities (insert at module level):
  ```rust
  use crate::device::{DeviceManager, GrabbedDevice};
  use aethermap_common::{DeviceCapabilities, DeviceType};

  /// Detect device capabilities from DeviceInfo and actual device
  async fn detect_device_capabilities(
      device_info: &aethermap_common::DeviceInfo,
      state: &DaemonState,
  ) -> DeviceCapabilities {
      // Check if device is grabbed (has evdev access)
      let has_analog = if let Some(grabbed) = state.device_manager.read().await
          .grabbed_devices.get(&device_info.path.to_string_lossy().to_string())
      {
          // Query actual device for absolute axes
          grabbed.evdev.supported_absolute_axes()
              .map_or(false, |axes| axes.iter().next().is_some())
      } else {
          // Fallback: infer from device type
          matches!(device_info.device_type, DeviceType::Keypad | DeviceType::Gamepad)
      };

      // Azeron keypads have hat switches
      let has_hat_switch = matches!(device_info.device_type, DeviceType::Keypad);

      // Button count: Azeron has 26 joystick buttons
      let joystick_button_count = if matches!(device_info.device_type, DeviceType::Keypad) {
          26
      } else {
          0
      };

      // LED zones: defer to Phase 12, empty for now
      let led_zones = Vec::new();

      DeviceCapabilities {
          has_analog_stick: has_analog,
          has_hat_switch,
          joystick_button_count,
          led_zones,
      }
  }
  ```

  Note: LED zones left empty - will be populated in Phase 12 (LED Control).
