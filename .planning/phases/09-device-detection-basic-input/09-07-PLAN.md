  /// Get device capabilities and features
  pub async fn get_device_capabilities(
      &self,
      device_path: &str,
  ) -> Result<DeviceCapabilities, String> {
      let request = Request::GetDeviceCapabilities {
          device_path: device_path.to_string(),
      };

      self.send_request(&request).await?;

      match self.receive_response().await? {
          Response::DeviceCapabilities { capabilities, .. } => Ok(capabilities),
          Response::Error(e) => Err(e),
          other => Err(format!("Unexpected response: {:?}", other)),
      }
  }
  ```

  /// View for Azeron keypad remapping interface
  fn view_azeron_keypad(&self) -> Element<Message> {
      let layout = azeron_keypad_layout();

      // Create grid of buttons
      let mut rows: Vec<Vec<Element<Message>>> = vec![vec![]; 10];

      for button in &layout {
          let button_id = button.id.clone();
          let label = button.label.clone();
          let remap = button.current_remap.clone();

          let btn = container(
              button(text(label))
                  .on_press(Message::SelectKeypadButton(button_id.clone()))
          )
          .padding(5)
          .style(if remap.is_some() {
              theme::Container::Custom(Box::new(|theme| {
                  container::Appearance {
                      background: Some(iced::Color::from_rgb(0.2, 0.6, 0.2)),
                      border_radius: 4.0.into(),
                      ..Default::default()
                  }
              }))
          } else {
              theme::Container::Box
          })
          .into();

          if rows.get_mut(button.row).is_some() {
              rows[button.row].push(btn);
          }
      }

      // Add hat switch in center
      let hat_switch = container(
          text("HAT").size(20)
      )
      .width(iced::Length::Fixed(60.0))
      .height(iced::Length::Fixed(60.0))
      .center_x()
      .center_y()
      .style(theme::Container::Box)
      .into();
      rows[4].push(hat_switch);  // Center row

      column(rows.into_iter().map(|row| row.into()).collect()).into()
  }
  ```
