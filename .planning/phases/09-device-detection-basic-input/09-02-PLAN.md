  3. Add a new match arm for AbsoluteAxis BEFORE the final "_ => {}" catch-all:

  ```rust
  match event.kind() {
      // Existing Key handling...
      InputEventKind::Key(key) => { /* existing code */ }

      // Existing RelAxis handling...
      InputEventKind::RelAxis(axis) => { /* existing code */ }

      // NEW: Handle absolute axis events (analog stick)
      InputEventKind::AbsoluteAxis(axis) => {
          let value = event.value();

          debug!("Event from {}: axis={:?}, value={}", path, axis, value);

          // Encode absolute axis events in the message format
          // We use high key codes to represent analog events:
          // 61000: ABS_X, 61001: ABS_Y, 61002: ABS_Z, 61003: ABS_RX, 61004: ABS_RY, 61005: ABS_RZ
          let analog_event_code = match axis {
              evdev::AbsoluteAxisType::ABS_X => 61000,
              evdev::AbsoluteAxisType::ABS_Y => 61001,
              evdev::AbsoluteAxisType::ABS_Z => 61002,
              evdev::AbsoluteAxisType::ABS_RX => 61003,
              evdev::AbsoluteAxisType::ABS_RY => 61004,
              evdev::AbsoluteAxisType::ABS_RZ => 61005,
              _ => {
                  debug!("Unsupported absolute axis: {:?}", axis);
                  continue; // Skip unsupported axes
              }
          };

          let sender_clone = sender.clone();
          let path_clone = path.clone();

          if let Err(e) = rt.block_on(sender_clone.send((path_clone, analog_event_code, value))) {
              error!("Failed to send analog event: {}", e);
              return;
          }
      }

      // Existing catch-all...
      _ => {}
  }
  ```

  4. Add the necessary import if not present (should already be imported via evdev):
  ```rust
  use evdev::AbsoluteAxisType;
  ```
