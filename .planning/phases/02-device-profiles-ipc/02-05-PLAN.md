    ```rust
    /// Activate a remap profile for a device
pub async fn activate_profile(
    &mut self,
    device_path: &str,
    profile_name: &str,
) -> Result<(), Box<dyn std::error::Error>> {
    // Get the grabbed device
    let grabbed_device = self.grabbed_devices.get_mut(device_path)
        .ok_or_else(|| format!("Device not grabbed: {}", device_path))?;

    // Get device key for profile lookup
    let device_key = format!("{:04x}:{:04x}",
        grabbed_device.info.vendor_id,
        grabbed_device.info.product_id
    );

    // Find the profile in stored profiles
    let profiles = self.device_profiles.get(&device_key)
        .ok_or_else(|| format!("No profiles found for device {}", device_key))?;

    let profile = profiles.iter()
        .find(|p| p.profile_name == profile_name)
        .ok_or_else(|| format!("Profile '{}' not found for device {}", profile_name, device_key))?;

    // Store the profile's remap table for quick access
    grabbed_device.profile_remaps.insert(profile_name.to_string(), profile.remap_table());
    grabbed_device.active_profile = Some(profile_name.to_string());

    info!("Activated profile '{}' for device {}", profile_name, device_path);
    Ok(())
}

/// Deactivate the current remap profile for a device
pub async fn deactivate_profile(
    &mut self,
    device_path: &str,
) -> Result<(), Box<dyn std::error::Error>> {
    let grabbed_device = self.grabbed_devices.get_mut(device_path)
        .ok_or_else(|| format!("Device not grabbed: {}", device_path))?;

    // Clear active profile
    grabbed_device.active_profile = None;

    info!("Deactivated profile for device {}", device_path);
    Ok(())
}

/// Get active remaps for a device
pub async fn get_active_remaps(
    &self,
    device_path: &str,
) -> Result<Option<(String, Arc<RemapTable>)>, Box<dyn std::error::Error>> {
    let grabbed_device = self.grabbed_devices.get(device_path)
        .ok_or_else(|| format!("Device not grabbed: {}", device_path))?;

    if let Some(profile_name) = &grabbed_device.active_profile {
        if let Some(table) = grabbed_device.profile_remaps.get(profile_name) {
            return Ok(Some((profile_name.clone(), Arc::clone(table))));
        }
    }

    Ok(None)
}
```

       ```rust
       Request::GetActiveRemaps { device_path } => {
           let state = state.read().await;

           let device_manager = state.device_manager.as_ref()
               .ok_or_else(|| Response::Error("Device manager not initialized".to_string()))?;

           let dm = device_manager.read().await;

           match dm.get_active_remaps(&device_path).await {
               Ok(Some((profile_name, remaps)) => {
                   // Convert remaps to Vec<RemapEntry>
                   let entries: Vec<RemapEntry> = remaps.iter()
                       .map(|(from, to)| RemapEntry {
                           from: format!("{:?}", from),
                           to: format!("{:?}", to),
                       })
                       .collect();

                   Response::ActiveRemaps {
                       device_path,
                       profile_name: Some(profile_name),
                       remaps: entries,
                   }
               }
               Ok(None) => {
                   Response::ActiveRemaps {
                       device_path,
                       profile_name: None,
                       remaps: vec![],
                   }
               }
               Err(e) => Response::Error(format!("Failed to get active remaps: {}", e)),
           }
       }
       ```

    2. ListRemapProfiles handler:
       ```rust
       Request::ListRemapProfiles { device_path } => {
           let state = state.read().await;

           // Get device info to determine device_id
           let device_info = match get_device_info_from_path(&device_path) {
               Some(info) => info,
               None => return Response::Error(format!("Device not found: {}", device_path)),
           };

           let device_id = format!("{:04x}:{:04x}", device_info.vendor_id, device_info.product_id);

           let device_manager = state.device_manager.as_ref()
               .ok_or_else(|| Response::Error("Device manager not initialized".to_string()))?;

           let dm = device_manager.read().await;

           match dm.get_device_profiles(&device_id) {
               Some(profiles) => {
                   let profile_names: Vec<String> = profiles.iter()
                       .map(|p| p.profile_name.clone())
                       .collect();

                   Response::RemapProfiles {
                       device_path,
                       profiles: profile_names,
                   }
               }
               None => Response::RemapProfiles {
                   device_path,
                   profiles: vec![],
               },
           }
       }
       ```
