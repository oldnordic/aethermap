       ```rust
       /// A pre-validated remap profile for a device
       ///
       /// Contains device identifier, profile name, and pre-parsed remap table.
       /// The remaps field is an Arc<RemapTable> allowing atomic profile switching
       /// without rebuilding HashMaps.
       #[derive(Debug, Clone)]
       pub struct RemapProfile {
           /// Device identifier (vendor:product, e.g., "1532:0220")
           pub device_id: String,
           /// Profile name (e.g., "default", "gaming")
           pub profile_name: String,
           /// Human-readable description
           pub description: Option<String>,
           /// Pre-validated remap table (shared Arc for atomic switching)
           pub remaps: Arc<RemapTable>,
       }

       impl RemapProfile {
           /// Create a new RemapProfile from parsed remaps
           pub fn new(
               device_id: String,
               profile_name: String,
               description: Option<String>,
               remaps: HashMap<evdev::Key, evdev::Key>,
           ) -> Self {
               Self {
                   device_id,
                   profile_name,
                   description,
                   remaps: Arc::new(remaps),
               }
           }

           /// Get the remap table as Arc for cloning
           pub fn remap_table(&self) -> Arc<RemapTable> {
               Arc::clone(&self.remaps)
           }
       }
       ```
