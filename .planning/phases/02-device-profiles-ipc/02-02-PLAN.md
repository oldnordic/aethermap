       ```rust
       #[derive(Debug, Clone, Serialize, Deserialize)]
       pub struct DeviceRemapConfig {
           /// Device match pattern (vendor:product, e.g., "1532:0220")
           #[serde(default)]
           pub match_pattern: Option<String>,

           /// Named profiles for this device
           #[serde(default)]
           pub profiles: HashMap<String, ProfileRemaps>,
       }

       /// Remap rules for a named profile
       #[derive(Debug, Clone, Serialize, Deserialize)]
       pub struct ProfileRemaps {
           /// Human-readable name
           #[serde(default)]
           pub name: String,

           /// Human-readable description
           #[serde(default)]
           pub description: Option<String>,

           /// Key remappings
           #[serde(default)]
           pub remaps: HashMap<String, String>,
       }

       /// Complete remap configuration with per-device profiles
       #[derive(Debug, Clone, Serialize, Deserialize)]
       pub struct RemapDevicesConfig {
           /// Per-device remap configurations keyed by device_id (vendor:product)
           #[serde(default)]
           pub devices: HashMap<String, DeviceRemapConfig>,

           /// Global default remaps (applied when no device-specific profile)
           #[serde(default)]
           pub default: Option<HashMap<String, String>>,
       ```

       ```rust
       /// Load per-device remap profiles from YAML configuration
       pub async fn load_device_profiles(&self) -> Result<HashMap<String, Vec<RemapProfile>>, RemapConfigError> {
           use crate::remap_engine::RemapProfile;
           use crate::key_parser::KeyParser;

           if !self.remap_devices_path.exists() {
               warn!("Remap devices config not found at {}, creating empty file", self.remap_devices_path.display());
               let empty = RemapDevicesConfig {
                   devices: HashMap::new(),
                   default: None,
               };
               let yaml = serde_yaml::to_string(&empty)?;
               tokio::fs::write(&self.remap_devices_path, yaml).await?;
               return Ok(HashMap::new());
           }

           let content = tokio::fs::read_to_string(&self.remap_devices_path).await
               .map_err(|e| RemapConfigError::ReadError {
                   path: self.remap_devices_path.clone(),
                   source: e,
               })?;

           let config: RemapDevicesConfig = serde_yaml::from_str(&content)
               .map_err(|e| RemapConfigError::ParseError {
                   path: self.remap_devices_path.clone(),
                   source: e,
               })?;

           let key_parser = KeyParser::new();
           let mut result: HashMap<String, Vec<RemapProfile>> = HashMap::new();

           // Process each device's profiles
           for (device_id, device_config) in &config.devices {
               let mut profiles = Vec::new();

               for (profile_name, profile_config) in &device_config.profiles {
                   let mut remap_table = HashMap::new();

                   // Validate and convert remaps
                   for (from, to) in &profile_config.remaps {
                       let from_key = key_parser.parse(from).map_err(|_| RemapConfigError::InvalidKey {
                           path: self.remap_devices_path.clone(),
                           key: from.clone(),
                           expected: format!("valid evdev key in device={}, profile={}", device_id, profile_name),
                       })?;

                       let to_key = key_parser.parse(to).map_err(|_| RemapConfigError::InvalidKey {
                           path: self.remap_devices_path.clone(),
                           key: to.clone(),
                           expected: format!("valid evdev key in device={}, profile={}", device_id, profile_name),
                       })?;

                       remap_table.insert(from_key, to_key);
                   }

                   profiles.push(RemapProfile::new(
                       device_id.clone(),
                       profile_name.clone(),
                       profile_config.description.clone(),
                       remap_table,
                   ));
               }

               result.insert(device_id.clone(), profiles);
           }

           info!("Loaded {} device profiles from {}", result.len(), self.remap_devices_path.display());
           Ok(result)
       }
       ```
