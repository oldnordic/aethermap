```rust
Request::GetActiveLayer { device_id } => {
    let state = state.read().await;
    let layer_manager = state.layer_manager.read().await;
    let effective_layer = layer_manager.get_effective_layer(&device_id);
    // Get layer name from DeviceLayerState
    let layer_name = layer_manager.get_device_state(&device_id)
        .and_then(|s| s.layer_configs.get(effective_layer))
        .map(|c| c.name.clone())
        .unwrap_or_else(|| format!("Layer {}", effective_layer));
    Response::ActiveLayer {
        device_id,
        layer_id: effective_layer,
        layer_name,
    }
}

Request::SetLayerConfig { device_id, layer_id, config } => {
    let mut state = state.write().await;
    let mut layer_manager = state.layer_manager.write().await;
    // Apply config to layer_manager
    match layer_manager.set_layer_config(&device_id, layer_id, config) {
        Ok(_) => Response::LayerConfigured { device_id, layer_id },
        Err(e) => Response::Error(format!("Failed to set layer config: {}", e)),
    }
}

Request::ActivateLayer { device_id, layer_id, mode } => {
    let mut state = state.write().await;
    let mut layer_manager = state.layer_manager.write().await;
    let result = match mode {
        LayerMode::Hold => layer_manager.activate_hold_layer(&device_id, layer_id),
        LayerMode::Toggle => {
            layer_manager.toggle_layer(&device_id, layer_id).map(|_| ())
        }
    };
    match result {
        Ok(_) => Response::LayerConfigured { device_id, layer_id },
        Err(e) => Response::Error(format!("Failed to activate layer: {}", e)),
    }
}

Request::ListLayers { device_id } => {
    let state = state.read().await;
    let layer_manager = state.layer_manager.read().await;
    let layers = layer_manager.get_device_state(&device_id)
        .map(|s| {
            s.layer_configs.iter().map(|c| LayerConfigInfo {
                layer_id: c.layer_id,
                name: c.name.clone(),
                mode: c.mode,
                remap_count: c.remaps.len(),
            }).collect()
        })
        .unwrap_or_default();
    Response::LayerList { device_id, layers }
}
```

3. Add set_layer_config() method to LayerManager (in layer_manager.rs):
```rust
pub fn set_layer_config(&mut self, device_id: &str, layer_id: usize, config: LayerConfigInfo) -> Result<(), String> {
    let device_state = self.get_or_create_device_state(device_id);
    if layer_id >= device_state.layer_configs.len() {
        return Err(format!("Invalid layer_id: {}", layer_id));
    }
    device_state.layer_configs[layer_id].name = config.name;
    device_state.layer_configs[layer_id].mode = config.mode;
    // Note: remaps updated separately via RemapEngine
    Ok(())
}
```
