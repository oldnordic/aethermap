3. Update AnalogInputUpdated handler to throttle:
   ```rust
   Message::AnalogInputUpdated(x, y) => {
       if let Some(view) = &mut self.analog_calibration_view {
           // Throttle to ~30 FPS (33ms between updates)
           if view.last_visualizer_update.elapsed() >= std::time::Duration::from_millis(33) {
               view.stick_x = x;
               view.stick_y = y;
               view.last_visualizer_update = Instant::now();
               Command::none() // Triggers redraw
           } else {
               Command::none() // Skip redraw, no state change
           }
       } else {
           Command::none()
       }
   }
   ```

       }
       Command::none()

Note: 30 FPS = 33ms provides smooth feedback without overwhelming the GUI. This is a UI best practice.
