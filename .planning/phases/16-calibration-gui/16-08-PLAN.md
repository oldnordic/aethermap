   ```rust
   #[cfg(test)]
   mod tests {
       use super::*;
       use aethermap_common::SensitivityCurve;

       #[test]
       fn test_apply_curve_linear() {
           // Linear: output = input
           assert!((apply_curve(0.0, &SensitivityCurve::Linear, 1.0) - 0.0).abs() < f32::EPSILON);
           assert!((apply_curve(0.5, &SensitivityCurve::Linear, 1.0) - 0.5).abs() < f32::EPSILON);
           assert!((apply_curve(1.0, &SensitivityCurve::Linear, 1.0) - 1.0).abs() < f32::EPSILON);
       }

       #[test]
       fn test_apply_curve_quadratic() {
           // Quadratic: output = input^2
           let result = apply_curve(0.5, &SensitivityCurve::Quadratic, 1.0);
           assert!((result - 0.25).abs() < 0.001);
       }

       #[test]
       fn test_apply_curve_exponential() {
           // Exponential: output = input^exponent
           let curve = SensitivityCurve::Exponential { exponent: 3.0 };
           let result = apply_curve(0.5, &curve, 1.0);
           assert!((result - 0.125).abs() < 0.001); // 0.5^3 = 0.125
       }

       #[test]
       fn test_apply_curve_with_multiplier() {
           // Multiplier scales output
           let result = apply_curve(0.5, &SensitivityCurve::Linear, 2.0);
           assert!((result - 1.0).abs() < 0.001); // 0.5 * 2.0 = 1.0
       }

       #[test]
       fn test_apply_curve_clamping() {
           // Output clamped to [-1.0, 1.0]
           let result = apply_curve(0.8, &SensitivityCurve::Linear, 2.0);
           assert_eq!(result, 1.0); // 0.8 * 2.0 = 1.6, clamped to 1.0
       }

       #[test]
       fn test_apply_curve_negative_input() {
           // Negative inputs handled correctly
           let result = apply_curve(-0.5, &SensitivityCurve::Quadratic, 1.0);
           assert!((result - (-0.25)).abs() < 0.001); // -0.5^2 = -0.25
       }
   }
   ```

These tests verify the curve math is correct, which is critical for accurate calibration visualization.
