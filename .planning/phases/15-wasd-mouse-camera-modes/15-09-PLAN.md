    2. When updating layer config, set `analog_mode` from `calibration.analog_mode` instead of preserving existing
    3. Extract `calibration.camera_output_mode` if present and set it when mode is Camera

    The pattern to follow:
    ```rust
    Request::SetAnalogCalibration { device_id, layer_id, calibration } => {
        // Save calibration
        let result = config_manager.save_analog_calibration(&device_id, layer_id, &calibration).await;

        // Also update analog_mode in layer config
        let layer_mode = calibration.analog_mode;
        let camera_mode = calibration.camera_output_mode.unwrap_or(CameraOutputMode::Scroll);

        let mut state = state.write().await;
        let mut layer_manager = state.layer_manager.write().await;

        // Get existing config or create default
        let existing_config = layer_manager.get_device_state(&device_id).await
            .and_then(|s| s.get_layer_config(layer_id));

        let mut layer_config = existing_config.cloned().unwrap_or_else(|| {
            LayerConfig::new(layer_id)
        });

        // Update mode from calibration
        layer_config.analog_mode = layer_mode;
        if layer_mode == AnalogMode::Camera {
            layer_config.camera_output_mode = camera_mode;
        }

        // Apply the updated config
        match layer_manager.set_layer_config(&device_id, layer_id, layer_config).await {
            Ok(_) => Response::AnalogCalibrationAck,
            Err(e) => Response::Error(format!("Failed to set analog mode: {}", e)),
        }
    }
    ```
