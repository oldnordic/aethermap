  <files>razermapper/razermapperd/src/device.rs</files>
  <action>
    Add Camera mode handling in the start_event_reader function, following the pattern established for Mouse mode (plan 15-04).

    Location: After the Mouse mode check from plan 15-04.

    Implementation pattern:
    1. Check if active layer uses Camera mode:
       ```rust
    if analog_mode == crate::analog_processor::AnalogMode::Camera {
        // Track both axes for camera processing
        match axis {
            AbsoluteAxisType::ABS_X => {
                dpad_state.x = value;
            }
            AbsoluteAxisType::ABS_Y => {
                dpad_state.y = value;

                // Process camera output when we have both axes
                let injector_clone = Arc::clone(injector);
                let lm_clone = Arc::clone(lm);
                let id_clone = id.clone();
                let analog_processor_clone = analog_processor.clone();

                rt.block_on(async move {
                    if let (Some(processor), Some(inj), Some(lm)) =
                        (&analog_processor_clone, &injector_clone, &lm_clone)
                    {
                        // Get layer-specific calibration and camera mode
                        let lm_read = lm_clone.read().await;
                        let device_state = lm_read.get_device_state(&id_clone).await;
                        let layer_id = device_state.as_ref()
                            .map(|s| s.get_effective_layer())
                            .unwrap_or(0);

                        let layer_config = device_state
                            .and_then(|s| s.get_layer_config(layer_id));

                        let layer_calibration = layer_config
                            .and_then(|c| c.analog_calibration().cloned());

                        let camera_mode = layer_config
                            .map(|c| c.camera_output_mode())
                            .unwrap_or(crate::analog_processor::CameraOutputMode::Scroll);

                        drop(lm_read);

                        // Get calibration
                        let calibration = if let Some(cal) = layer_calibration {
                            cal
                        } else {
                            // Create default calibration
                            use crate::analog_calibration::{AnalogCalibration, DeadzoneShape, SensitivityCurve};
                            AnalogCalibration {
                                deadzone: 0.15,
                                deadzone_shape: DeadzoneShape::Circular,
                                sensitivity: SensitivityCurve::Linear,
                                sensitivity_multiplier: 1.0,
                                range_min: -32768,
                                range_max: 32767,
                                invert_x: false,
                                invert_y: false,
                            }
                        };

                        // Process as camera
                        if let Some(output) = processor.process_as_camera(
                            &calibration,
                            dpad_state.x,
                            dpad_state.y,
                            camera_mode,
                        ) {
                            match output {
                                crate::analog_processor::CameraOutput::Scroll(amount) => {
                                    let _ = inj.mouse_scroll(amount).await;
                                    debug!("Camera scroll: device={}, amount={}", id_clone, amount);
                                }
                                crate::analog_processor::CameraOutput::Keys(keys) => {
                                    // Emit keys for camera control
                                    for key in keys {
                                        let _ = inj.key_press(key.0).await;
                                        let _ = inj.key_release(key.0).await;  // Immediate for repeat
                                    }
                                    debug!("Camera keys: device={}, keys={:?}", id_clone, keys);
                                }
                            }
                        }
                    }
                });

                // Skip sending to macro engine when in camera mode
                continue;
            }
            _ => {}
        }
    }
       ```

    Note: Key mode emits immediate press/release for repeat effect. Proper repeat interval handling can be added later if needed.

    DO NOT create new state tracking - camera events are fire-and-forget like mouse events.
